function getScrollTop(){  
	var scrollTop=0;
	if(document.documentElement&&document.documentElement.scrollTop){  
		scrollTop=document.documentElement.scrollTop;
	}else if(document.body){  
		scrollTop=document.body.scrollTop;  
	}  
	return scrollTop;
}

function getClientHeight(){  
	var clientHeight=0;  
	if(document.body.clientHeight&&document.documentElement.clientHeight){  
		var clientHeight = (document.body.clientHeight<document.documentElement.clientHeight)?document.body.clientHeight:document.documentElement.clientHeight;
	}else  {  
		var clientHeight = (document.body.clientHeight>document.documentElement.clientHeight)?document.body.clientHeight:document.documentElement.clientHeight;      
	}  

	return clientHeight;  
}  

function getScrollHeight(){  
	return Math.max(document.body.scrollHeight,document.documentElement.scrollHeight);  
}

function clone(myObj){
	if(typeof(myObj) != 'object') return myObj;
	if(myObj == null) return myObj;
	var myNewObj = new Object();
	for(var i in myObj)
		myNewObj[i] = myObj[i];//clone(myObj[i]);
	return myNewObj;
}

<p>　　中新网12月14日电 据中国国防科技信息网报道，美国海军新型濒海战斗舰用来探测水雷的主要设备——远程猎雷系统经过多次试验后，于12日被证实存有明显的性能缺陷。美国海军在过去1年中对该设备的研发总共投入了大约7亿美元。</p>
<p>　　美国国防部的武器试验评估部门证实了远程猎雷系统存在性能缺陷。去年9月进行的试验中，该系统在海中对水雷的探测和破坏能力就没有完全达标。美国国防部主管采购的副部长肯德尔表示，将在2016年初对远程猎雷系统的研发工作进行讨论。</p>
<p>　　该系统由美国洛克希德·马丁公司开发。在研发计划中，水雷由濒海战斗舰在目标海域中布放，当系统在海中探测到水雷时，将相关信息发送至濒海战斗舰，然后引导载舰回避或者破坏水雷。</p>
<p>　　据美国海军武器试验评估部门去年的年度报告书显示，远程猎雷系统试验的失败曾引起国会议员的批评。虽然今年进行了多次性能试验，但该系统仍旧没能得到有效的数据统计信息，性能状况和10年前的水平相似。</p>
<p>　　在试验中，从濒海战斗舰中将远程猎雷系统远距离投放的作业总是不能顺利操作，通信也不能顺利实现。因此，该系统对水雷探测能力的持续性遭到质疑。</p>
<p>　　根据美国联邦参议院军事委员会今年9月的文件，远程猎雷系统的期待工作时间为75小时，但是在试验中最长持续工作时间仅约为25小时。</p>
<p>　　洛克希德·马丁公司声明，将继续推进远程猎雷系统的研发。今后将继续与美国海军保持密切的合作。该公司同时声称，该系统将会是美国海军舰队装备的最先进的水雷探测装置。</p>
